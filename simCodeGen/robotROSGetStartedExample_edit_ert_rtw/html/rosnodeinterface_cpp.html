135    void NodeInterface::rosClockSubscriberCallback(const rosgraph_msgs::Clock &



136      msg)



137    {



138      static bool termFlag = false;



139      static bool waitingFlag = false;



140      static int stateFlag = 1;



141      const uint64_t baseRateNanoSec = 10000000;



142      static uint64_t nextTimeSlot = baseRateNanoSec;



143      uint64_t longTime;



144      if ((1 == stateFlag) && (0U != msg.clock.nsec) ) {



145        ROS_INFO("Waiting for time synchronization at time %u.000000000",



146                 msg.clock.sec + 1U);



147      }



148



149      if (false != termFlag) {



150        return;



151      }



152



153      longTime = ((uint64_t)msg.clock.sec * 1000000000ULL) + (uint64_t)



154        msg.clock.nsec;



155



156      // Check if time fits into model steps



157      if ((0 == stateFlag) && (true == waitingFlag) ) {



158        if (longTime > nextTimeSlot) {



159          fflush(stdout);



160          ROS_FATAL("The clock missed time slot at %f for the model step. Actual clock = %.6f\n",



161                    nextTimeSlot/1E9, (double)longTime/1E9);



162          fflush(stderr);



163          mRunModel = false;



164          mBaseRateSem.notify();



165          termFlag = true;



166          return;



167        }



168      }



169



170      // Store into global variable published by clock topic



171      _clock_msg_sec = msg.clock.sec;



172      _clock_msg_nsec = msg.clock.nsec;



173



174      // Starting the model code at time of "nsec" at 0ns



175      if (false == waitingFlag) {



176        if (0U == msg.clock.nsec) {



177          waitingFlag = true;



178          ROS_INFO("** Unblocking base-rate at %.6f", (double)longTime/1E9);



179          modelRuns = true;



180          mBaseRateSem.notify();       // Launch for first time



181          nextTimeSlot = longTime + baseRateNanoSec;



182        } else {



183          // Check if there was no termination in the meantime



184          mRunModel = (rtmGetErrorStatus(mModel->getRTM()) == (NULL));



185          if (mRunModel == false) {



186            termFlag = true;



187          }



188        }



189      } else {



190        if ((msg.clock.nsec % baseRateNanoSec) == 0U) {



191          // Overrun check



192          if (false != modelRuns) {



193            numOverruns++;



194            ROS_ERROR("Overrun %d", numOverruns);



195          }



196



197          modelRuns = true;



198          mBaseRateSem.notify();



199          nextTimeSlot += baseRateNanoSec;



200        }



201      }



202



203      if (stateFlag > 0) {



204        stateFlag--;



205      }



206    }

211    void NodeInterface::schedulerThread(void)



212    {



213



214#ifdef MW_DEBUG_LOG



215



216      ROS_INFO("ROS schedulerTask entered\n");



217



218#endif



219



220      std::string sClockTopic = "/clock";



221      ros::NodeHandle n;



222      ros::Subscriber sub = n.subscribe(sClockTopic, 10, &ros::matlab::



223        NodeInterface::rosClockSubscriberCallback, this);



224      if (sub == NULL) {



225        ROS_INFO("Error: cannot subscribe the '%s'", sClockTopic.c_str());



226        return;



227      }



228



229      ros::spin();



230



231      // Unblock main thread and terminate



232      mStopSem.notify();



233      return;



234    }

237    void NodeInterface::baseRateTask(void)



238    {



239      mRunModel = (rtmGetErrorStatus(mModel->getRTM()) ==



240                   (NULL));



241      while (mRunModel) {



242        mBaseRateSem.wait();



243



244#ifdef MW_DEBUG_LOG



245



246        ROS_INFO("** Base rate task semaphore received\n");



247



248#endif



249



250        if (!mRunModel)



251          break;



252        mModel->step();



253        mRunModel = !NodeInterface::getStopRequestedFlag();



254      }



255



256      // Shutdown the ROS node



257      ros::shutdown();



258    }
